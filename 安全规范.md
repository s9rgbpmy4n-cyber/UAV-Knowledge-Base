# ============================================================
# 核心函数
# ============================================================

def ensure_dir(path: Path) -> None:
    """确保目录存在，不存在则创建。"""
    path.mkdir(parents=True, exist_ok=True)


def get_md_files() -> list[Path]:
    """
    获取 docs/ 目录下所有 .md 文件，按文件名排序。

    Returns:
        list[Path]: Markdown 文件路径列表
    """
    if not DOCS_DIR.exists():
        return []
    return sorted(DOCS_DIR.glob("*.md"))


def read_file(path: Path) -> str:
    """
    读取文本文件内容。

    Args:
        path: 文件路径

    Returns:
        str: 文件内容
    """
    try:
        return path.read_text(encoding="utf-8")
    except Exception as e:
        print(f"[警告] 读取文件失败: {path} - {e}")
        return ""


def md_to_html(md_content: str) -> str:
    """
    将 Markdown 内容转换为 HTML。

    支持的扩展：
        - tables: 表格支持
        - fenced_code: 围栏式代码块
        - toc: 目录生成
        - nl2br: 换行转 <br>
        - sane_lists: 更合理的列表处理

    Args:
        md_content: Markdown 原始文本

    Returns:
        str: 转换后的 HTML 片段
    """
    md = markdown.Markdown(extensions=[
        "tables",
        "fenced_code",
        "toc",
        "nl2br",
        "sane_lists",
    ])
    html = md.convert(md_content)
    return html


def build_nav_links(current_file: str, md_files: list[Path]) -> str:
    """
    构建导航栏链接 HTML。

    Args:
        current_file: 当前页面文件名（如 "index.html"）
        md_files: docs/ 目录下的所有 .md 文件列表

    Returns:
        str: 导航链接的 HTML 字符串
    """
    links = []

    # 首页链接
    active_class = "active" if current_file == "index.html" else ""
    links.append(f'<a href="index.html" class="{active_class}">首页</a>')

    # 文档页面链接
    for md_file in md_files:
        html_name = md_file.stem + ".html"
        title = md_file.stem  # 使用文件名（不含扩展名）作为标题
        active_class = "active" if current_file == html_name else ""
        links.append(f'<a href="{html_name}" class="{active_class}">{title}</a>')

    return "\n".join(f"<li>{link}</li>" for link in links)


def generate_page(title: str, description: str, body_html: str,
                  current_file: str, md_files: list[Path]) -> str:
    """
    生成完整的 HTML 页面。

    Args:
        title: 页面标题
        description: 页面描述（meta description）
        body_html: 正文 HTML 内容
        current_file: 当前页面文件名
        md_files: 用于导航栏的 .md 文件列表

    Returns:
        str: 完整的 HTML 页面
    """
    nav_links = build_nav_links(current_file, md_files)

    return HTML_TEMPLATE.format(
        title=title,
        description=description,
        project_title=PROJECT_TITLE,
        nav_links=nav_links,
        body_content=body_html,
        primary_color=PRIMARY_COLOR,
        primary_dark=PRIMARY_DARK,
        primary_light=PRIMARY_LIGHT,
        bg_color=BG_COLOR,
        text_color=TEXT_COLOR,
        border_color=BORDER_COLOR,
        AUTHOR=AUTHOR,
        github_url=GITHUB_URL,
    )


def write_html(output_path: Path, content: str) -> None:
    """
    将 HTML 内容写入文件。

    Args:
        output_path: 输出文件路径
        content: HTML 内容
    """
    try:
        output_path.write_text(content, encoding="utf-8")
        print(f"[成功] 已生成: {output_path}")
    except Exception as e:
        print(f"[错误] 写入文件失败: {output_path} - {e}")


def process_readme(md_files: list[Path]) -> None:
    """
    处理 README.md，生成首页 index.html。

    步骤：
        1. 读取 README.md 内容
        2. 修正内部链接（.md → .html）
        3. 转换为 HTML
        4. 生成完整页面

    Args:
        md_files: 文档文件列表（用于导航栏）
    """
    print("\n[1/2] 处理 README.md → index.html")

    md_content = read_file(README_FILE)
    if not md_content:
        print("[警告] README.md 为空或不存在，跳过首页生成")
        return

    # 修正 README 中的链接：docs/xxx.md → xxx.html
    # 将相对路径的 .md 链接替换为 .html
    md_content = re.sub(
        r'href="docs/([^"]+)\.md"',
        r'href="\1.html"',
        md_content
    )

    # 转换 Markdown → HTML
    body_html = md_to_html(md_content)

    # 生成完整 HTML 页面
    html = generate_page(
        title="首页",
        description=PROJECT_DESC,
        body_html=body_html,
        current_file="index.html",
        md_files=md_files,
    )

    # 写入文件
    output_path = SITE_DIR / "index.html"
    write_html(output_path, html)


def process_docs(md_files: list[Path]) -> None:
    """
    处理 docs/ 目录下的所有 .md 文件，逐个转换为 .html。

    Args:
        md_files: Markdown 文件路径列表
    """
    total = len(md_files)
    print(f"\n[2/2] 处理 docs/ 目录 ({total} 个文件)")

    for i, md_file in enumerate(md_files, 1):
        print(f"  ({i}/{total}) {md_file.name} → {md_file.stem}.html")

        # 读取 Markdown 内容
        md_content = read_file(md_file)
        if not md_content:
            continue

        # 转换 Markdown → HTML
        body_html = md_to_html(md_content)

        # 生成完整 HTML 页面
        html_name = md_file.stem + ".html"
        html = generate_page(
            title=md_file.stem,
            description=f"{PROJECT_DESC} - {md_file.stem}",
            body_html=body_html,
            current_file=html_name,
            md_files=md_files,
        )

        # 写入文件
        output_path = SITE_DIR / html_name
        write_html(output_path, html)


def build() -> None:
    """
    主构建函数，执行完整的站点构建流程。

    流程：
        1. 确保 site/ 输出目录存在
        2. 获取 docs/ 目录下的所有 .md 文件
        3. 处理 README.md → site/index.html
        4. 处理 docs/*.md → site/*.html
        5. 输出构建统计信息
    """
    print("=" * 60)
    print(f" {PROJECT_TITLE} - 静态站点构建")
    print(f" {AUTHOR}")
    print("=" * 60)

    # 步骤1：确保输出目录存在
    ensure_dir(SITE_DIR)
    print(f"\n[准备] 输出目录: {SITE_DIR.absolute()}")

    # 步骤2：获取所有 Markdown 文件
    md_files = get_md_files()
    print(f"[准备] 发现 {len(md_files)} 个 Markdown 文件")
    for f in md_files:
        print(f"       - {f.name}")

    # 步骤3：处理 README.md 生成首页
    process_readme(md_files)

    # 步骤4：处理 docs/ 下的所有文件
    process_docs(md_files)

    # 步骤5：构建统计
    print("\n" + "=" * 60)
    print(" 构建完成!")
    print("=" * 60)

    html_files = list(SITE_DIR.glob("*.html"))
    print(f" 生成文件总数: {len(html_files)}")
    for f in sorted(html_files):
        size = f.stat().st_size
        print(f"   - {f.name} ({size:,} bytes)")

    print(f"\n 请在浏览器中打开以下文件预览:")
    print(f"   {SITE_DIR.absolute()}/index.html")
    print("=" * 60)


# ============================================================
# 程序入口
# ============================================================

if __name__ == "__main__":
    build()
